<script>
  // Data setup
  const competencies = [
    // Social (8)
    ["Interpersonal Communication & Collaboration","social"],
    ["Flexibility & Coping with Ambiguity","social"],
    ["Self-Regulation & Empathy","social"],
    ["Critical Thinking & Problem Framing","social"],
    ["Future Literacy","social"],
    ["Communication & Presentation Skills","social"],
    ["Recognizing Opportunities, Valuing Ideas & Creativity","social"],
    ["Motivation & Goal Orientation","social"],
    // Organisational (10)
    ["Mobilizing Resources","organizational"],
    ["Taking Initiative","organizational"],
    ["Long-Term Planning","organizational"],
    ["Engagement & Collective Empowerment","organizational"],
    ["Creation of Learning Opportunities by Visionary Design","organizational"],
    ["Strategic Education Marketing","organizational"],
    ["Policy Implementation & Manage Funding","organizational"],
    ["Promoting & Valuing Ecological & Social Sustainability","organizational"],
    ["Understanding Economic & Financial Concepts","organizational"],
    ["Financial Planning & Budgeting","organizational"],
    // Subject Knowledge (7)
    ["Managing Data, Information & Digital Content","subject"],
    ["Evaluating Relevance & Validity of Data & Digital Content","subject"],
    ["Creating Digital Content","subject"],
    ["Communication & Collaboration through Digital Tools","subject"],
    ["Safety & Keeping Things Up to Date","subject"],
    ["Understanding & Promoting Smart Technologies","subject"],
    ["Managing Smart Technology Implementation & Integration","subject"]
  ];
  const thresholds = {
    social:[8,13,14,19,20,24],
    organizational:[10,16,17,24,25,30],
    subject:[7,11,12,17,18,21]
  };
  const maxScore={social:24,organizational:30,subject:21};
  const fieldLabels={social:"Social",organizational:"Organisational",subject:"Subject Knowledge"};
  const guidance={
    basic:["üî∞","Basic","Foundational level‚Äîbuild confidence."],
    intermediate:["‚ö°","Intermediate","Solid skills‚Äîfocus on refinement."],
    advanced:["üèÜ","Advanced","Mastery‚Äîready to lead."]
  };
  const moduleLinks={social:[2,3,4],organizational:[3,4],subject:[1,3,4]};

  let answers=Array(25).fill(null), idx=0;

  // Build question cards
  const qContainer=document.getElementById("questions");
  competencies.forEach(([text,field],i)=>{
    const card=document.createElement("div");
    card.className="question-card";
    card.id="qcard"+i;
    card.innerHTML=`
      <div>
        <h5>Question ${i+1} of ${competencies.length}</h5>
        <p>${text}</p>
        <select id="q${i}" class="form-control">
          <option value="">Select‚Ä¶</option>
          <option value="0">0 ‚Äì Not at all</option>
          <option value="1">1 ‚Äì A little</option>
          <option value="2">2 ‚Äì Moderately</option>
          <option value="3">3 ‚Äì Very well</option>
        </select>
      </div>`;
    qContainer.appendChild(card);
  });
  document.getElementById("qcard0").classList.add("active");

  // Wizard controls
  function updateWizard(){
    const pct=Math.round((idx/competencies.length)*100);
    document.getElementById("progress-bar").style.width=pct+"%";
    document.getElementById("prev-btn").disabled=idx===0;
    document.getElementById("next-btn").textContent= idx===competencies.length-1?"Submit":"Next";
    document.querySelectorAll(".question-card")
      .forEach((c,i)=>c.classList.toggle("active",i===idx));
  }
  document.getElementById("next-btn").onclick=()=>{
    const sel=document.getElementById("q"+idx).value;
    if(sel==="") return alert("Please select an option.");
    answers[idx]=+sel;
    if(idx<competencies.length-1){ idx++; updateWizard(); }
    else showResults();
  };
  document.getElementById("prev-btn").onclick=()=>{
    if(idx>0) { idx--; updateWizard(); }
  };

  // Show results
  function showResults(){
    const sums={social:0,organizational:0,subject:0};
    competencies.forEach(([,f],i)=>sums[f]+=answers[i]);
    document.querySelector(".wizard-container").classList.add("d-none");
    const charts=document.getElementById("charts");
    Object.entries(sums).forEach(([field,total])=>{
      const [b1,b2,i1,i2,a1,a2]=thresholds[field];
      let cat="basic";
      if(total>=a1) cat="advanced";
      else if(total>=i1) cat="intermediate";
      const [emoji,label,text]=guidance[cat];
      const pct=Math.round(total/maxScore[field]*100);
      const chart=document.createElement("div");
      chart.innerHTML=`
        <div class="radial-chart">
          <div class="mask"></div>
          <div class="fill" style="transform:rotate(${pct*3.6}deg)"></div>
          <div class="inside-circle">${pct}%</div>
        </div>
        <h5>${fieldLabels[field]}</h5>
        <h6>${emoji} ${label}</h6>
        <p>${text}</p>
        <div class="module-links">
          ${moduleLinks[field].map(i=>`<a href="https://ruraled.eu/toolkit/module${i}" target="_blank">Module ${i}</a>`).join("")}
        </div>`;
      charts.appendChild(chart);
    });
    document.getElementById("results").classList.remove("d-none");
    document.getElementById("results").scrollIntoView({behavior:"smooth"});
  }
</script>
